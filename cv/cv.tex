%%  \cvperson{Full Name}{Email}{Phone}{LinkedIn handle}{GitHub handle}{Website}
\newcommand{\cvperson}[6]{%
  \begin{center}
    {\LARGE\bfseries\color{accent} #1}\\[4pt]
    {\small
      \href{mailto:#2}{#2}
      \enspace\textbar\enspace
      #3
      \enspace\textbar\enspace
      \href{https://www.linkedin.com/in/italo-salgado-910764192/}{linkedin.com/in/#4}
      \enspace\textbar\enspace
      \href{https://github.com/#5}{github.com/#5}%
      \ifx&#6&\else
        \enspace\textbar\enspace\href{https://#6}{#6}%
      \fi
    }
  \end{center}
}

%%  \experience{Job Title}{Company}{Location}{Date Range}{Bullet list of achievements}
%%  Achievements: use \item inside a standard itemize (see examples below).
\newcommand{\experience}[5]{%
  \noindent
  \begin{tabularx}{\linewidth}{@{}X r@{}}
    \textbf{#1} \textit{at} \textbf{#2}, #3 & \textit{#4}
  \end{tabularx}
  \vspace{1pt}
  #5
  \vspace{4pt}
}

%%  \project{Project Name}{Tech Stack}{Date Range}{Description bullet list}
\newcommand{\project}[4]{%
  \noindent
  \begin{tabularx}{\linewidth}{@{}X r@{}}
    \textbf{#1} \hspace{4pt}{\small\textit{#2}} & \textit{#3}
  \end{tabularx}
  \vspace{1pt}
  #4
  \vspace{4pt}
}

%%  \education{Degree}{Institution}{Location}{Date Range}{Details / GPA / honours}
\newcommand{\education}[5]{%
  \noindent
  \begin{tabularx}{\linewidth}{@{}X r@{}}
    \textbf{#1} — \textit{#2}, #3 & \textit{#4}
  \end{tabularx}
  \ifx&#5&\else{\small #5}\fi
  \vspace{4pt}
}

%%  \skillgroup{Category}{Comma-separated skill list}
\newcommand{\skillgroup}[2]{%
  \noindent\textbf{#1:} #2\par\vspace{2pt}
}
