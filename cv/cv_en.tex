\newcommand{\cvperson}[6]{%
  \begin{center}
    {\LARGE\bfseries\color{accent} #1}\\[4pt]
    {\small
      \href{mailto:#2}{#2}
      \enspace\textbar\enspace
      #3
      \enspace\textbar\enspace
      \href{#4}{LinkedIn}
      \enspace\textbar\enspace
      \href{#5}{GitHub}%
      \ifx&#6&\else
        \enspace\textbar\enspace\href{#6}{Website}%
      \fi
    }
  \end{center}
}

\newcommand{\experience}[5]{%
  \noindent
  \begin{tabularx}{\linewidth}{@{}X r@{}}
    \textbf{#1} \textit{at} \textbf{#2}, #3 & \textit{#4}
  \end{tabularx}
  \vspace{-10pt}
  #5
  \vspace{2pt}
}

    \textbf{#1} \hspace{4pt}{\small\textit{#2}} & \textit{#3}
  \end{tabularx}
  \vspace{0pt}
  #4
  \vspace{0pt}
}

    \textbf{#1} — \textit{#2}, #3 & \textit{#4}
  \end{tabularx}
  \ifx&#5&\else{\small #5}\fi
  \vspace{0pt}
}
